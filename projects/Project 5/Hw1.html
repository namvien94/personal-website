import numpy as np
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression

np.random.seed(2)

def true_function(x):
    """The true underlying function: sin(x) + cos(x)."""
    return np.sin(x) + np.cos(x)

def generate_data(n_datasets, n_samples, noise_std=0.20):
    """Generates multiple datasets with noise."""
    x = np.random.uniform(-2 * np.pi, 2 * np.pi, 
                          size=(n_datasets, n_samples))
    noise = np.random.normal(0, noise_std, 
                             size=(n_datasets, n_samples))
    y = true_function(x) + noise
    return x, y

def generate_test_data(n_samples):
    """Generates test data without noise."""
    x_test = np.random.uniform(-2 * np.pi, 2 * np.pi, size=n_samples)
    y_test = true_function(x_test)
    return x_test, y_test

def create_polynomial_features(x, degree):
    """Manually creates a polynomial feature matrix 
    up to a given degree."""
    X_poly = np.ones((len(x), degree + 1))
    for i in range(1, degree + 1):
        X_poly[:, i] = x ** i
    return X_poly

def fit_polynomial_model(x_train, y_train, x_test, degree):
    """Fits polynomial regression models for each dataset 
    and returns predictions on test data."""
    n_datasets = x_train.shape[0]
    y_pred_all = []
    X_test_poly = create_polynomial_features(x_test, degree)
    for i in range(n_datasets):
        X_train_poly = create_polynomial_features(x_train[i], degree)
        model = LinearRegression()
        model.fit(X_train_poly, y_train[i])
        y_pred_all.append(model.predict(X_test_poly))
    return np.array(y_pred_all)

def estimate_bias_variance(y_pred_all, y_test):
    """Estimates the squared bias and variance for a given model."""
    y_pred_mean = np.mean(y_pred_all, axis=0)
    bias_squared = np.mean((y_pred_mean - y_test) ** 2)
    variance_estimate = np.mean(np.var(y_pred_all, axis=0))
    return bias_squared, variance_estimate

def plot_bias_variance_tradeoff(degrees, squared_bias, variance):
    """Plots the bias-variance tradeoff."""
    plt.figure(figsize=(8, 6))
    plt.plot(degrees, squared_bias, label="Squared Bias", 
             color='blue', marker='o')
    plt.plot(degrees, variance, label="Variance", 
             color='orange', marker='o')
    plt.xlabel("Degrees of polynomial")
    plt.ylabel("Value")
    plt.legend(loc='best')
    plt.grid(True)
    plt.title("Bias-Variance Tradeoff")
    plt.show()

def main():
    n_datasets = 1000
    n_samples_per_dataset = 30
    n_test_samples = 50000
    degrees = np.arange(1, 8)

    x_train, y_train = generate_data(n_datasets, n_samples_per_dataset)
    x_test, y_test = generate_test_data(n_test_samples)

    squared_bias = []
    variance = []

    for degree in degrees:
        y_pred_all = fit_polynomial_model(x_train, y_train, x_test, degree)
        bias_squared, variance_estimate = estimate_bias_variance(y_pred_all, y_test)
        squared_bias.append(bias_squared)
        variance.append(variance_estimate)

    plot_bias_variance_tradeoff(degrees, squared_bias, variance)

if __name__ == "__main__":
    main()

import numpy as np
import matplotlib.pyplot as plt
from sklearn.neighbors import KNeighborsClassifier
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score

np.random.seed(246010)

def generate_custom_data(n_samples):
    """
    Generates (x1, x2) from a uniform distribution [-1,1].
    Sets the probability P(Y = 1) using the given formula 
    and generates binary labels.
    """
    X = np.random.uniform(-1, 1, (n_samples, 2))
    prob_y = 1 / (1 + np.exp(-X[:, 0] - X[:, 1]))
    y = np.random.binomial(1, prob_y)
    return X, y

# 2. Train and test KNN with different values of K
def knn_test(X_train, y_train, X_test, y_test, k_values):
    test_errors = []
    for k in k_values:
        knn = KNeighborsClassifier(n_neighbors=k)
        knn.fit(X_train, y_train)
        y_pred = knn.predict(X_test)
        test_error = 1 - accuracy_score(y_test, y_pred)
        test_errors.append(test_error)
    return test_errors

# 3. Plot the testing error as a function of K
def plot_knn_test_error(k_values, test_errors):
    plt.figure(figsize=(8, 6))
    plt.plot(k_values, test_errors, marker='o', 
             linestyle='-', color='orange')
    plt.xlabel("Number of K")
    plt.ylabel("Testing Error")
    plt.title("Testing Error vs K")
    plt.grid(True)
    plt.show()

# Main function
def main():
    # Generate the custom dataset with 300 samples for training 
    # and 10,000 for testing
    X_train, y_train = generate_custom_data(300)
    X_test, y_test = generate_custom_data(10000)

    # Define values of K up to 13 as shown in the image
    k_values = [3, 5, 7, 9, 11, 13]

    # Compute testing error for each value of K
    test_errors = knn_test(X_train, y_train, X_test, y_test, k_values)

    # Plot the testing error fluctuation
    plot_knn_test_error(k_values, test_errors)

# Run the main function
if __name__ == "__main__":
    main()
