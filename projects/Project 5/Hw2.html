(1) Only consider two features X_1 = variance and X_2 = skewness.

Use logistic loss and a linear function class, and minimize the following task via gradient descent (you may refer to the supplementary file of Week 2 for detailed implementations):

$$ \frac{1}{n} \sum_{i=1}^{n} \log(1 + \exp(\beta_0 + \beta_1 \cdot \text{variance} + \beta_2 \cdot \text{skewness})) $$

Once you estimate the parameters, plot the decision boundary.

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.preprocessing import StandardScaler

# Load dataset from GitHub URL
data_url = 'https://github.com/zzh237/stat101C/blob/main/BankNote_Authentication.csv?raw=true'
data = pd.read_csv(data_url)

# Select only the two features: variance and skewness, along with the target variable
X = data[['variance', 'skewness']].values
y = data['class'].values
y = 2 * y - 1  # Convert y to -1 and 1 for consistency with the supplementary document

# Standardize features for consistency in model performance
scaler = StandardScaler()
X = scaler.fit_transform(X)

# Initialize parameters
beta_0 = 0.0
beta_12 = np.array([0.0, 0.0])  # for beta_1 and beta_2
learning_rate = 0.05
iterations = 30000

# Gradient descent
for i in range(iterations):
    Temp = (X @ beta_12 + beta_0) * y
    Temp_2 = np.exp(-Temp) / (1 + np.exp(-Temp)) * y
    beta_0 += learning_rate * np.mean(Temp_2)
    beta_12[0] += learning_rate * np.mean(Temp_2 * X[:, 0])
    beta_12[1] += learning_rate * np.mean(Temp_2 * X[:, 1])

# Final parameters
print(f"Estimated Parameters: beta_0 = {beta_0}, beta_1 = {beta_12[0]}, beta_2 = {beta_12[1]}")

# Plot decision boundary
plt.figure(figsize=(10, 6))
plt.scatter(X[y == -1][:, 0], X[y == -1][:, 1], color='blue', label='Class -1')
plt.scatter(X[y == 1][:, 0], X[y == 1][:, 1], color='red', label='Class 1')

# Calculate decision boundary
x_vals = np.linspace(X[:, 0].min(), X[:, 0].max(), 100)
y_vals = -(beta_0 + beta_12[0] * x_vals) / beta_12[1]
plt.plot(x_vals, y_vals, label='Decision Boundary', color='black', linestyle='--')

plt.xlabel('Variance (Standardized)')
plt.ylabel('Skewness (Standardized)')
plt.legend()
plt.title('Decision Boundary for Logistic Regression')
plt.show()

# Select only the two features and the target variable
X = data[['variance', 'skewness']].values
y = data['class'].values

# Standardize features
scaler = StandardScaler()
X = scaler.fit_transform(X)

from sklearn.discriminant_analysis import LinearDiscriminantAnalysis

# Initialize and train the LDA model
lda = LinearDiscriminantAnalysis()
lda.fit(X, y)

# Plotting decision boundary for LDA
plt.figure(figsize=(10, 6))
# Create a mesh grid over the feature space
x_min, x_max = X[:, 0].min() - 1, X[:, 0].max() + 1
y_min, y_max = X[:, 1].min() - 1, X[:, 1].max() + 1
xx, yy = np.meshgrid(np.arange(x_min, x_max, 0.01),
                     np.arange(y_min, y_max, 0.01))

# Predict on the mesh grid
Z = lda.predict(np.c_[xx.ravel(), yy.ravel()])
Z = Z.reshape(xx.shape)

# Plot decision boundary and data points
plt.contourf(xx, yy, Z, alpha=0.3, cmap='coolwarm')
plt.scatter(X[y == 0][:, 0], X[y == 0][:, 1], color='blue', label='Class 0')
plt.scatter(X[y == 1][:, 0], X[y == 1][:, 1], color='red', label='Class 1')

plt.xlabel('Variance (Standardized)')
plt.ylabel('Skewness (Standardized)')
plt.legend()
plt.title('Problem 2: LDA Decision Boundary')
plt.show()
