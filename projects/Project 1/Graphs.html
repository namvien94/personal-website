import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.cluster import DBSCAN, MeanShift, AgglomerativeClustering, estimate_bandwidth

# Load data
latlon = pd.read_csv('latitude_longitude_with_clusters.csv')
X = latlon[['LATITUDE','LONGITUDE']].values

# Filter US bounds
mask = (X[:,0] > 15) & (X[:,0] < 55) & (X[:,1] > -130) & (X[:,1] < -60)
X_filtered = X[mask]

# Sample for performance
np.random.seed(42)
sample_idx = np.random.choice(len(X_filtered), size=min(3000, len(X_filtered)), replace=False)
X_sample = X_filtered[sample_idx]

#################################
# 1. DBSCAN
#################################
db = DBSCAN(eps=0.3, min_samples=5).fit(X_sample)
db_labels = db.labels_

unique_labels = sorted(set(db_labels))
cmap = plt.cm.tab20
colors = [cmap(i % 20) for i in range(len(unique_labels))]
color_map = {lbl: (0.8,0.8,0.8,1) if lbl == -1 else colors[i] for i, lbl in enumerate(unique_labels)}
db_colors = [color_map[lbl] for lbl in db_labels]

plt.figure(figsize=(9,7))
plt.scatter(X_sample[:,1], X_sample[:,0], c=db_colors, s=10)
plt.title("DBSCAN", fontsize=18)
plt.xlabel("Longitude")
plt.ylabel("Latitude")
plt.grid(True, linestyle='--', alpha=0.4)
plt.tight_layout()
db_path = "DBSCAN.png"
plt.savefig(db_path, dpi=300)
plt.close()

#################################
# 2. Mean Shift
#################################
bandwidth = estimate_bandwidth(X_sample, quantile=0.2)
reduced_bw = bandwidth * 0.4
ms = MeanShift(bandwidth=reduced_bw, bin_seeding=True).fit(X_sample)
ms_labels = ms.labels_

unique_labels = sorted(set(ms_labels))
colors = [cmap(i % 20) for i in range(len(unique_labels))]
color_map = {lbl: colors[i] for i, lbl in enumerate(unique_labels)}
ms_colors = [color_map[lbl] for lbl in ms_labels]

plt.figure(figsize=(9,7))
plt.scatter(X_sample[:,1], X_sample[:,0], c=ms_colors, s=10)
plt.title("Mean Shift", fontsize=18)
plt.xlabel("Longitude")
plt.ylabel("Latitude")
plt.grid(True, linestyle='--', alpha=0.4)
plt.tight_layout()
ms_path = "MeanShift.png"
plt.savefig(ms_path, dpi=300)
plt.close()

#################################
# 3. Agglomerative Clustering
#################################
agg = AgglomerativeClustering(n_clusters=10).fit(X_sample)
agg_labels = agg.labels_

unique_labels = sorted(set(agg_labels))
colors = [cmap(i % 20) for i in range(len(unique_labels))]
color_map = {lbl: colors[i] for i, lbl in enumerate(unique_labels)}
agg_colors = [color_map[lbl] for lbl in agg_labels]

plt.figure(figsize=(9,7))
plt.scatter(X_sample[:,1], X_sample[:,0], c=agg_colors, s=10)
plt.title("Agglomerative Clustering", fontsize=18)
plt.xlabel("Longitude")
plt.ylabel("Latitude")
plt.grid(True, linestyle='--', alpha=0.4)
plt.tight_layout()
agg_path = "Agglomerative.png"
plt.savefig(agg_path, dpi=300)
plt.close()

import matplotlib.image as mpimg

for path, title in [(db_path, "DBSCAN"), (ms_path, "Mean Shift"), (agg_path, "Agglomerative Clustering")]:
    img = mpimg.imread(path)
