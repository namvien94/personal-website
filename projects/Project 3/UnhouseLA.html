# @title CALENDAR_YEAR vs count()

from matplotlib import pyplot as plt
import seaborn as sns
def _plot_series(series, series_name, series_index=0):
  palette = list(sns.palettes.mpl_palette('Dark2'))
  counted = (series['CALENDAR_YEAR']
                .value_counts()
              .reset_index(name='counts')
              .rename({'index': 'CALENDAR_YEAR'}, axis=1)
              .sort_values('CALENDAR_YEAR', ascending=True))
  xs = counted['CALENDAR_YEAR']
  ys = counted['counts']
  plt.plot(xs, ys, label=series_name, color=palette[series_index % len(palette)])

fig, ax = plt.subplots(figsize=(10, 5.2), layout='constrained')
df_sorted = data2.sort_values('CALENDAR_YEAR', ascending=True)
for i, (series_name, series) in enumerate(df_sorted.groupby('GENDER_CATEGORY')):
  _plot_series(series, series_name, i)
  fig.legend(title='GENDER_CATEGORY', bbox_to_anchor=(1, 1), loc='upper left')
sns.despine(fig=fig, ax=ax)
plt.xlabel('CALENDAR_YEAR')
_ = plt.ylabel('count()')

Cisgender Men and Cisgender Women consistently have the highest counts, with steady increases over the years, particularly in 2024. Transgender individuals and those identifying as culturally specific or non-binary also show gradual growth, while the "Questioning" category has a noticeable rise starting in 2020. The "Client doesn't know" category maintains moderate growth. These trends indicate increasing representation and reporting across all categories over time. By analyzing these trends, we can infer potential shifts in service provision patterns over the years, which may correlate with changes in neighborhood-level service allocations. This graph helps contextualize which demographic groups might drive the demand for services in specific areas, offering indirect insights into neighborhoods consistently receiving higher or lower levels of homeless response services.

# Load the dataset
data_path = "data1.csv"  # Replace with your file path
data = pd.read_csv(data_path)

# Drop rows with missing latitude or longitude values
data = data.dropna(subset=['latitude', 'longitude'])

# Create a map centered on Los Angeles
map_with_cities = folium.Map(location=[34.0522, -118.2437], zoom_start=10)

# Group data by city ('name') and add markers for each city
for city, group in data.groupby('name'):
    # Extract city latitude, longitude, and create a popup message with year and homeless count info
    latitude = group['latitude'].iloc[0]
    longitude = group['longitude'].iloc[0]
    popup_message = "<b>{}</b><br>".format(city)
    popup_message += "<br>".join(
        [f"Year: {row['year']}, Homeless Count: {row['tothomeless_count']}" for _, row in group.iterrows()]
    )

    # Add a marker to the map
    folium.Marker(
        location=[latitude, longitude],
        popup=folium.Popup(popup_message, max_width=300)
    ).add_to(map_with_cities)

# Let the map object be the last line to display it
map_with_cities
